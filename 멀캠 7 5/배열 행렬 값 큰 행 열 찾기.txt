/*
배열에서 각 요소에 정수가 들어 있다. 이 배열에서 합이 가장 큰 행과 열의 번호를 각각 인쇄하라.
아래와 같이 배열이 주어지며 주어진 배열을 이용하여 구현하다.
int a[4][5] = { { 3, -5, 12, 3, -21 },{ -2, 11, 2, -7, -11 }, { 21, -21, -35, -93, -11 },{ 9, 14, 39, -98, -1 } };



각 행의 합을 구해보면 가장 합이 큰 행은 2행으로 -2 + 11 + 2 + -7 + -11 = -7 합계이다.
각 열의 합을 구하면 가장 합이 큰 열은 1열로 3 + -2 + 21 + 9 = 31 합계이다.
따라서 합이 가장 큰 행번호와 열번호는 2행과 1열이다.
단, 행과 열의 번호는 0이 아닌 1부터 시작하고 있다.

*/

#include<stdio.h>




int main()
{
	int max = -100000000;
	int temp = 0;
	int x, y;
	int i;
	int j;

	int a[4][5] = {
		{ 3, -5, 12, 3, -21 },
		{ -2, 11, 2, -7, -11 },
		{ 21, -21, -35, -93, -11 },
		{ 9, 14, 39, -98, -1 }
	};

	for (i = 0; i < 4; i++)
	{
		for (j = 0; j < 5; j++)
		{
			temp += a[i][j];
		}
	
		if (max < temp) {
			max = temp; 
			x = i;
		}
		temp = 0;

		
	}
	printf("%d", x+1);
	max = -10000000;

	for (i = 0; i < 5; i++)
	{
		for (j = 0; j < 4; j++)
		{
			temp += a[j][i];
		}

		if (max < temp) {
			max = temp;
			y = i;
		}
		temp = 0;


	}

	printf("%d", y+1);



	return 0;
}
